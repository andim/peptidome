```python
nt_to_ind = {
    'A' : 0,
    'C' : 1,
    'G' : 2,
    'U' : 3 
    }
def calc_frequencies(p):
    frequencies = {aa:0 for aa in df.index}
    for nts, aa in codon_map.items():
        if not aa == 'STOP':
            frequencies[aa] += np.prod([p[nt_to_ind[nt]] for nt in nts])
    return frequencies
```
